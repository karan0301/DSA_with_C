int arr[]={67,4,7,82};
void bubbleSort(int arr[])
{
    for (int i = 0; i < 4-1; i++)
    {
        for (int j = 0; j < 4-i-1; j++)
        {
            if (arr[j]>arr[j+1])
            {
                int temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
}
